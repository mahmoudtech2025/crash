    <div class="container">
        <h1>لعبة Crash Simulation</h1>
        <p id="multiplier">1.00x</p>
        <p id="message">لقد انهارت اللعبة! 😢</p>
        <button class="btn" onclick="startGame()">ابدأ اللعبة</button>
        <button class="btn" onclick="cashOut()">صرف الأرباح</button>
    </div>

    <script>
        let multiplier = 1.00;
        let crashPoint = Math.random() * (5.0 - 1.5) + 1.5;
        let running = false;
        let interval;

        function startGame() {
            running = true;
            document.getElementById("message").style.display = "none";
            multiplier = 1.00;
            crashPoint = Math.random() * (5.0 - 1.5) + 1.5;

            interval = setInterval(() => {
                multiplier += 0.01;
                document.getElementById("multiplier").innerText = multiplier.toFixed(2) + "x";

                if (multiplier >= crashPoint) {
                    clearInterval(interval);
                    document.getElementById("message").style.display = "block";
                    running = false;
                }
            }, 100);
        }

        function cashOut() {
            if (running) {
                alert("لقد ربحت: " + multiplier.toFixed(2) + "x");
                clearInterval(interval);
                running = false;
            } else {
                alert("انتهت الجولة! حاول مرة أخرى.");
            }
        }
    </script>
